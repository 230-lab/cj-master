let currentText = "";

let currentIndex = 0;

let startTime = null;

let timerId = null;

let mistakes = 0;


const textDisplay = document.getElementById('textDisplay');

const typingInput = document.getElementById('typingInput');

const articleSelect = document.getElementById('articleSelect');


// 初始化選單

articles.forEach((art, i) => {

    let opt = document.createElement('option');

    opt.value = i;

    opt.innerText = art.title;

    articleSelect.appendChild(opt);

});


function loadArticle(idx) {

    currentText = articles[idx].content;

    currentIndex = 0;

    mistakes = 0;

    startTime = null;

    if(timerId) clearInterval(timerId);

    

    document.getElementById('timeVal').innerText = "0";

    document.getElementById('wpmVal').innerText = "0";

    document.getElementById('mistakeVal').innerText = "0";

    renderText();

    updateHint();

    typingInput.value = "";

}


function renderText() {

    textDisplay.innerHTML = "";

    currentText.split("").forEach((char, i) => {

        let span = document.createElement('span');

        span.innerText = char;

        span.className = "char " + (i < currentIndex ? "char-done" : (i === currentIndex ? "char-active" : ""));

        textDisplay.appendChild(span);

    });

    document.getElementById('progressVal').innerText = `${currentIndex}/${currentText.length}`;

}


function updateHint() {

    const char = currentText[currentIndex];

    const code = cjDict[char];

    if (!code) {

        document.getElementById('codeHint').innerText = "請輸入空白鍵跳過標點符號";

    } else {

        const chineseCode = code.split("").map(l => keyMap[l] || l).join("");

        document.getElementById('codeHint').innerText = `${char}：${chineseCode} (${code.toUpperCase()})`;

    }

}


typingInput.addEventListener('input', (e) => {

    if(!startTime) {

        startTime = new Date();

        timerId = setInterval(updateStats, 1000);

    }


    const val = e.target.value.toLowerCase().trim();

    const char = currentText[currentIndex];

    const correctCode = cjDict[char];


    // 處理標點符號

    if (!correctCode) {

        currentIndex++;

        typingInput.value = "";

        checkFinish();

        return;

    }


    if (val === correctCode) {

        currentIndex++;

        typingInput.value = "";

        checkFinish();

    } else if (val.length >= correctCode.length) {

        mistakes++;

        document.getElementById('mistakeVal').innerText = mistakes;

        typingInput.value = "";

    }

});


function updateStats() {

    const elapsed = Math.floor((new Date() - startTime) / 1000);

    document.getElementById('timeVal').innerText = elapsed;

    if (elapsed > 0) {

        const wpm = Math.round((currentIndex / elapsed) * 60);

        document.getElementById('wpmVal').innerText = wpm;

    }

}


function checkFinish() {

    if (currentIndex >= currentText.length) {

        clearInterval(timerId);

        showResult();

    } else {

        renderText();

        updateHint();

    }

}

function closeModal() {
